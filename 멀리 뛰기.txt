//Lv2 멀리 뛰기
//피보나치 수열
class Solution {
    public long solution(int n) {
        long[] answer = new long[n + 1];
        if (n == 1)
            return 1;
        if (n == 2)
            return 2;
        answer[1] = 1;
        answer[2] = 2;
        for (int i = 3; i <= n; i++) {
            answer[i] = (answer[i - 1] + answer[i - 2]) % 1234567;
        }
        return answer[n];
    }
}
